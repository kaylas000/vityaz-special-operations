// backend/src/combat/combat.service.ts
@Injectable()
export class CombatService {
  private readonly logger = new Logger(CombatService.name);
  
  /**
   * Calculates damage based on weapon, distance, body part, and armor
   * @param params Damage calculation parameters
   * @returns Final damage value (0-100)
   */
  calculateDamage(params: DamageParams): number {
    let damage = params.baseDamage;
    
    // Distance falloff
    damage *= this.applyDistanceFalloff(params.distance);
    
    // Body part multiplier
    damage *= this.getBodyPartMultiplier(params.bodyPart);
    
    // Armor reduction
    damage *= (1 - Math.min(0.8, params.armor / 100));
    
    return Math.max(0, Math.min(100, Math.floor(damage)));
  }
}
